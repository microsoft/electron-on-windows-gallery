import { AIFeatureReadyState, TextRecognizer } from '#winapp/bindings';
async function recognizeText(imageBuffer) {
  const recognizer = await TextRecognizer.createAsync();
  try {
    if (TextRecognizer.getReadyState() !== AIFeatureReadyState.Ready) {
      await TextRecognizer.ensureReadyAsync();
    }
 
    const recognizedText =
      await recognizer.recognizeTextFromImageAsync(imageBuffer);
    const lines = recognizedText.lines;
 
    const resultArray = [];
    for (const line of lines) {
      const bb = line.boundingBox;
      resultArray.push({
        text: line.text,
        boundingBox: [bb.topLeft.x, bb.topLeft.y],
        height: Math.abs(bb.topRight.y - bb.bottomRight.y),
      });
    }
    return resultArray;
  } finally {
    recognizer.close();
  }
}
