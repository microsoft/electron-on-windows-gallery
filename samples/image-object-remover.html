import {
  ImageObjectRemover,
  SoftwareBitmap,
  ImageBuffer,
  BitmapPixelFormat,
} from '#winapp/bindings';
 
async function removeObject(
  imageBitmap,
  maskBitmap,
  signal = new AbortController().signal,
) {
  const remover = await ImageObjectRemover.createAsync(signal);
  try {
    // maskBitmap has white pixels over the selected rectangle
    const maskGray = SoftwareBitmap.convert(
      maskBitmap,
      BitmapPixelFormat.Gray8,
    );
    // synchronous — run in a utility process to keep the UI responsive
    const resultBitmap = remover.removeFromSoftwareBitmap(
      imageBitmap,
      maskGray,
    );
    const resultBuffer = ImageBuffer.createForSoftwareBitmap(resultBitmap);
    const stride = resultBuffer.rowStride;
    const height = resultBuffer.pixelHeight;
    const pixels = resultBuffer.copyToByteArray(
      new Uint8Array(stride * height),
    );
    return { width: resultBuffer.pixelWidth, height, stride, pixels };
  } finally {
    remover.close();
  }
}
